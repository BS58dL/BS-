local function ForcePart(v)
    if v:IsA("Part") and not v.Anchored and not v.Parent:FindFirstChild("Humanoid") and not v.Parent:FindFirstChild("Head") and v.Name ~= "Handle" then
        for _, x in next, v:GetChildren() do
            if x:IsA("BodyAngularVelocity") or x:IsA("BodyForce") or x:IsA("BodyGyro") or x:IsA("BodyPosition") or x:IsA("BodyThrust") or x:IsA("BodyVelocity") or x:IsA("RocketPropulsion") then
                x:Destroy()
            end
        end
        if v:FindFirstChild("Attachment") then
            v:FindFirstChild("Attachment"):Destroy()
        end
        if v:FindFirstChild("AlignPosition") then
            v:FindFirstChild("AlignPosition"):Destroy()
        end
        if v:FindFirstChild("Torque") then
            v:FindFirstChild("Torque"):Destroy()
        end
        v.CanCollide = false
        local Torque = Instance.new("Torque", v)
        Torque.Torque = Vector3.new(math.random(-500000,500000), math.random(-500000,500000), math.random(-500000,500000))
        local AlignPosition = Instance.new("AlignPosition", v)
        local Attachment2 = Instance.new("Attachment", v)
        Torque.Attachment0 = Attachment2
        AlignPosition.MaxForce = 9999999999999999
        AlignPosition.MaxVelocity = math.huge
        AlignPosition.Responsiveness = 200
        AlignPosition.Attachment0 = Attachment2
        AlignPosition.Attachment1 = Attachment1
    end
end

local dragging
local dragInput
local dragStart
local startPos

local function update(input)
    local delta = input.Position - dragStart
    MainFrame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
end

MainFrame.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

MainFrame.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        update(input)
    end
end)

local radius = 500
local height = 200
local rotationSpeed = 2.0
local attractionStrength = 2000
local chaosMode = false

local chaosStrength = 2000
local chaosFrequency = 0.3
local maxChaosLevel = 3
local currentChaosLevel = 3

local function RetainPart(Part)
    if Part:IsA("BasePart") and not Part.Anchored and Part:IsDescendantOf(workspace) then
        if Part.Parent == LocalPlayer.Character or Part:IsDescendantOf(LocalPlayer.Character) then
            return false
        end

        Part.CustomPhysicalProperties = PhysicalProperties.new(0, 0, 0, 0, 0)
        Part.CanCollide = false
        return true
    end
    return false
end

local parts = {}
local function addPart(part)
    if RetainPart(part) then
        if not table.find(parts, part) then
            table.insert(parts, part)
        end
    end
end

local function removePart(part)
    local index = table.find(parts, part)
    if index then
        table.remove(parts, index)
    end
end

for _, part in pairs(workspace:GetDescendants()) do
    addPart(part)
end

workspace.DescendantAdded:Connect(addPart)
workspace.DescendantRemoving:Connect(removePart)

local function applySuperChaosEffect(part, center, chaosLevel)
    if not chaosMode then return end
    
    local levelMultiplier = chaosLevel
    
    local randomRotation = Vector3.new(
        math.random(-chaosStrength * levelMultiplier, chaosStrength * levelMultiplier),
        math.random(-chaosStrength * levelMultiplier, chaosStrength * levelMultiplier),
        math.random(-chaosStrength * levelMultiplier, chaosStrength * levelMultiplier)
    )
    
    local randomOffset = Vector3.new(
        math.random(-50 * levelMultiplier, 50 * levelMultiplier),
        math.random(-50 * levelMultiplier, 50 * levelMultiplier),
        math.random(-50 * levelMultiplier, 50 * levelMultiplier)
    )
    
    local randomVelocity = Vector3.new(
        math.random(-200 * levelMultiplier, 200 * levelMultiplier),
        math.random(-200 * levelMultiplier, 200 * levelMultiplier),
        math.random(-200 * levelMultiplier, 200 * levelMultiplier)
    )
    
    if math.random() < chaosFrequency then
        part.RotVelocity = part.RotVelocity + randomRotation
        part.Velocity = part.Velocity + randomVelocity * 2
    end
    
    if math.random() < 0.1 then
        part.Velocity = part.Velocity + Vector3.new(
            math.random(-500 * levelMultiplier, 500 * levelMultiplier),
            math.random(-500 * levelMultiplier, 500 * levelMultiplier),
            math.random(-500 * levelMultiplier, 500 * levelMultiplier)
        )
    end
end

local function createVisualEffects()
end

RunService.Heartbeat:Connect(function()
    if not chaosMode then return end
    
    local humanoidRootPart = LocalPlayer.Character and LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    if humanoidRootPart then
        local tornadoCenter = humanoidRootPart.Position
        
        createVisualEffects()
        
        for _, part in pairs(parts) do
            if part.Parent and not part.Anchored then
                local pos = part.Position
                local distance = (pos - tornadoCenter).Magnitude
                
                if distance <= radius * 2 then
                    
                    applySuperChaosEffect(part, tornadoCenter, currentChaosLevel)
                    
                    local randomAngle1 = math.rad(math.random(0, 360))
                    local randomAngle2 = math.rad(math.random(0, 360))
                    local randomRadius = math.random(radius * 0.1, radius * 1.5)
                    
                    local targetPos = Vector3.new(
                        tornadoCenter.X + math.cos(randomAngle1) * randomRadius + math.random(-100, 100),
                        tornadoCenter.Y + math.sin(randomAngle2) * height + math.random(-50, 50),
                        tornadoCenter.Z + math.sin(randomAngle1) * randomRadius + math.random(-100, 100)
                    )
                    
                    local directionToTarget = (targetPos - part.Position).unit
                    
                    local randomAttraction = attractionStrength + math.random(-1000, 2000)
                    part.Velocity = directionToTarget * randomAttraction + Vector3.new(
                        math.random(-300, 300),
                        math.random(-300, 300),
                        math.random(-300, 300)
                    )
                    
                    part.RotVelocity = Vector3.new(
                        math.random(-100, 100) * currentChaosLevel,
                        math.random(-100, 100) * currentChaosLevel,
                        math.random(-100, 100) * currentChaosLevel
                    ) * 10
                    
                    if math.random() < 0.05 then
                        part.Velocity = part.Velocity * 3
                        part.RotVelocity = part.RotVelocity * 5
                    end
                end
            end
        end
    end
end)
