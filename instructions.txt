        local dragging = false
        local dragStart, startPos

        local function update(input)
            if not dragging then return end
            local delta
            if input.UserInputType == Enum.UserInputType.Touch then
                delta = input.Position - dragStart
            else
                delta = input.Position - dragStart
            end
            gui.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)

            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then

                dragStart = input.Position
                startPos = gui.Position

                    dragElement.BackgroundColor3 = Color3.fromRGB(60, 60, 80)
                end
            end
        end)

            if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                dragging = false
                if input.UserInputType == Enum.UserInputType.Touch then
                    dragElement.BackgroundColor3 = Color3.fromRGB(40, 40, 60)
                end
            end
        end)

            if (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) and dragging then

            end
        end)

function AuthorCommandGUI:PlayAnimation(command, buttonFrame)
    local character = self.LocalPlayer.Character
    if not character then
        self:ShowNotification("ÈîôËØØ", "ËßíËâ≤‰∏çÂ≠òÂú®", 3)
        return
    end

    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid then
        self:ShowNotification("ÈîôËØØ", "Êó†Ê≥ïÊâæÂà∞Humanoid", 3)
        return
    end

    local animator = humanoid:FindFirstChildOfClass("Animator")
    if not animator then
        self:ShowNotification("ÈîôËØØ", "Êó†Ê≥ïÊâæÂà∞Animator", 3)
        return
    end

    if self.CurrentAnimations[command.name] then
        self.CurrentAnimations[command.name]:Stop()
        self.CurrentAnimations[command.name] = nil
    end

    local animationId = command.animationIds[1]
    
    local animation = Instance.new("Animation")
    animation.AnimationId = animationId

    local animationTrack = animator:LoadAnimation(animation)
    animationTrack:Play()
    
    self.CurrentAnimations[command.name] = animationTrack

    local playButton = buttonFrame:FindFirstChild("PlayButton")
    local stopButton = buttonFrame:FindFirstChild("StopButton")
    
    if playButton and stopButton then
        playButton.Visible = false
        stopButton.Visible = true
    end

    self:ShowNotification("Êí≠ÊîæÂä®Áîª", "Ê≠£Âú®Êí≠Êîæ: " .. command.name, 2)

    animationTrack.Stopped:Connect(function()
        if playButton and stopButton then
            playButton.Visible = true
            stopButton.Visible = false
        end
        self.CurrentAnimations[command.name] = nil
    end)
end

function AuthorCommandGUI:StopAnimation(command)
    if self.CurrentAnimations[command.name] then
        self.CurrentAnimations[command.name]:Stop()
        self.CurrentAnimations[command.name] = nil
        self:ShowNotification("ÂÅúÊ≠¢Âä®Áîª", "Â∑≤ÂÅúÊ≠¢: " .. command.name, 2)
    end
end

function AuthorCommandGUI:MinimizeGUI()
    if self.MainGUI then
        local mainFrame = self.MainGUI:FindFirstChild("MainFrame")
        local quickIcon = self.MainGUI:FindFirstChild("QuickIcon")
        
        if mainFrame and quickIcon then
            mainFrame.Visible = false
            quickIcon.Visible = true
            self.IsMinimized = true
        end
    end
end

function AuthorCommandGUI:MaximizeGUI()
    if self.MainGUI then
        local mainFrame = self.MainGUI:FindFirstChild("MainFrame")
        local quickIcon = self.MainGUI:FindFirstChild("QuickIcon")
        
        if mainFrame and quickIcon then
            mainFrame.Visible = true
            quickIcon.Visible = true
            self.IsMinimized = false
        end
    end
end

function AuthorCommandGUI:ToggleGUI()
    if not self.MainGUI then
        self.MainGUI = self:CreateGUI()
        self:MaximizeGUI()
    else
        if self.IsMinimized then
            self:MaximizeGUI()
        else
            self:MinimizeGUI()
        end
    end
end

function AuthorCommandGUI:SetupHotkey()
    self.UserInputService.InputBegan:Connect(function(input, gameProcessed)
        if gameProcessed then return end
        
        if input.KeyCode == Enum.KeyCode.F8 then
            self:ToggleGUI()
        end
    end)
end
